<script type="importmap">
{
    "imports": {
        "three": "https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js",

<script type="module">
import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

let scene, camera, renderer, orbitControls, raycaster;
let playerGroup, avatar, floor;
let moveForward = false, moveBackward = false, moveLeft = false, moveRight = false;
let moveUp = false, moveDown = false;
let canJump = false, isFlyMode = false, isThirdPerson = false;

const objectsToTest = []; 
const stairs = []; // 계단 오브젝트들을 따로 관리하기 위한 배열
const joystickData = { active: false, x: 0, y: 0 };
const velocity = new THREE.Vector3();
const direction = new THREE.Vector3();
let prevTime = performance.now();
const PLAYER_HEIGHT = 1.6;

init();
setupEnvironment();
setupJoystick();
animate();

function init() {
    scene = new THREE.Scene();
    scene.background = new THREE.Color(0x87ceeb);
    scene.fog = new THREE.FogExp2(0x87ceeb, 0.001);

    camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 5000);
    renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    renderer.shadowMap.enabled = true;
    document.body.appendChild(renderer.domElement);

    raycaster = new THREE.Raycaster(new THREE.Vector3(), new THREE.Vector3(0, -1, 0), 0, 10);

    playerGroup = new THREE.Group();
    playerGroup.position.set(0, PLAYER_HEIGHT, 0);
    scene.add(playerGroup);

    orbitControls = new OrbitControls(camera, renderer.domElement);
    orbitControls.enableDamping = true;
    orbitControls.dampingFactor = 0.1;
    orbitControls.minDistance = 0.1;
    orbitControls.maxDistance = 15;
    orbitControls.enablePan = false;
    orbitControls.mouseButtons = {
        LEFT: null, 
        MIDDLE: THREE.MOUSE.DOLLY,
        RIGHT: THREE.MOUSE.ROTATE 
    };
    orbitControls.target.copy(playerGroup.position);

    const avatarGroup = new THREE.Group();
    avatarGroup.add(new THREE.Mesh(new THREE.CapsuleGeometry(0.3, 1.0), new THREE.MeshStandardMaterial({ color: 0x3b82f6 })));
    avatar = avatarGroup; avatar.position.y = -0.5; avatar.visible = false;
    playerGroup.add(avatar);

    scene.add(new THREE.AmbientLight(0xffffff, 0.7));
    const sun = new THREE.DirectionalLight(0xffffff, 1.5);
    sun.position.set(100, 200, 100); sun.castShadow = true;
    scene.add(sun);

    document.getElementById('view-mode-btn').addEventListener('click', toggleFlyMode);
    document.getElementById('camera-mode-btn').addEventListener('click', toggleCameraMode);
    document.addEventListener('keydown', onKeyDown);
    document.addEventListener('keyup', onKeyUp);
    window.addEventListener('resize', onWindowResize);
}

function setupEnvironment() {
    const loader = new THREE.TextureLoader();
    const grassMap = loader.load('https://threejs.org/examples/textures/terrain/grasslight-big.jpg');
    grassMap.wrapS = grassMap.wrapT = THREE.RepeatWrapping;
    grassMap.repeat.set(1000, 1000);
    floor = new THREE.Mesh(new THREE.PlaneGeometry(10000, 10000), new THREE.MeshStandardMaterial({ map: grassMap }));
    floor.rotation.x = -Math.PI / 2;
    floor.receiveShadow = true;
    scene.add(floor);
    objectsToTest.push(floor);

    // 테스트용 계단 생성
    for(let i=0; i<10; i++) {
        const step = new THREE.Mesh(
            new THREE.BoxGeometry(5, 0.3, 1),
            new THREE.MeshStandardMaterial({ color: 0x888888 })
        );
        step.position.set(0, 0.15 + (i * 0.3), -5 - i);
        step.receiveShadow = true;
        step.castShadow = true;
        scene.add(step);
        objectsToTest.push(step);
        stairs.push(step); // 계단 배열에 추가
    }
}

function setupJoystick() {
    const container = document.getElementById('joystick-container');
    const knob = document.getElementById('joystick-knob');
    const rect = container.getBoundingClientRect();
    const maxRadius = rect.width / 2;
    const handleTouch = (e) => {
        e.preventDefault();
        const touch = e.touches[0];
        const dx = touch.clientX - (rect.left + maxRadius);
        const dy = touch.clientY - (rect.top + maxRadius);
        const dist = Math.min(Math.sqrt(dx*dx + dy*dy), maxRadius);
        const angle = Math.atan2(dy, dx);
        const moveX = dist * Math.cos(angle);
        const moveY = dist * Math.sin(angle);
        knob.style.transform = `translate(calc(-50% + ${moveX}px), calc(-50% + ${moveY}px))`;
        joystickData.x = moveX / maxRadius; joystickData.y = moveY / maxRadius; joystickData.active = true;
    };
    container.addEventListener('touchstart', handleTouch);
    container.addEventListener('touchmove', handleTouch);
    container.addEventListener('touchend', () => { knob.style.transform = `translate(-50%, -50%)`; joystickData.active = false; });
}

function toggleFlyMode() { 
    isFlyMode = !isFlyMode; 
    document.getElementById('view-mode-btn').innerText = `비행: ${isFlyMode ? 'ON' : 'OFF'}`; 
    velocity.y = 0; 
}

function toggleCameraMode() { 
    isThirdPerson = !isThirdPerson; 
    avatar.visible = isThirdPerson; 
    document.getElementById('camera-mode-btn').innerText = `시점: ${isThirdPerson ? '3인칭' : '1인칭'}`; 
    
    if (isThirdPerson) {
        orbitControls.minDistance = 2;
        orbitControls.maxDistance = 15;
        camera.position.addScaledVector(new THREE.Vector3(0,0,1), 5);
    } else {
        orbitControls.minDistance = 0.1;
        orbitControls.maxDistance = 0.2;
    }
}

// 계단 가시성 토글 함수
function toggleStairs() {
    const isVisible = stairs.length > 0 ? !stairs[0].visible : true;
    stairs.forEach(step => {
        step.visible = isVisible;
    });
}

function onKeyDown(e) { 
    if(e.code === 'KeyW') moveForward = true; 
    if(e.code === 'KeyS') moveBackward = true;
    if(e.code === 'KeyA') moveLeft = true; 
    if(e.code === 'KeyD') moveRight = true;
    if(e.code === 'KeyE') moveUp = true; 
    if(e.code === 'KeyQ') moveDown = true;
    if(e.code === 'KeyV') toggleCameraMode();
    if(e.code === 'KeyI') toggleStairs(); // I 키로 계단 가시성 토글
    if(e.code === 'Space' && canJump && !isFlyMode) { velocity.y += 8; canJump = false; }
}

function onKeyUp(e) { 
    if(e.code === 'KeyW') moveForward = false; 
    if(e.code === 'KeyS') moveBackward = false;
    if(e.code === 'KeyA') moveLeft = false; 
    if(e.code === 'KeyD') moveRight = false;
    if(e.code === 'KeyE') moveUp = false; 
    if(e.code === 'KeyQ') moveDown = false;
}

function animate() {
    requestAnimationFrame(animate);
    const time = performance.now();
    const delta = Math.min((time - prevTime) / 1000, 0.1);

    velocity.x -= velocity.x * 10.0 * delta;
    velocity.z -= velocity.z * 10.0 * delta;
    if (!isFlyMode) velocity.y -= 9.8 * 2.5 * delta; 
    else velocity.y -= velocity.y * 5.0 * delta;

    let mX = (Number(moveRight) - Number(moveLeft)) || (joystickData.active ? joystickData.x : 0);
    let mZ = (Number(moveForward) - Number(moveBackward)) || (joystickData.active ? -joystickData.y : 0);
    direction.set(mX, 0, mZ); if (direction.length() > 0.1) direction.normalize();

    const speed = isFlyMode ? 600 : 350;
    velocity.x += direction.x * speed * delta;
    velocity.z += direction.z * speed * delta;
    if (isFlyMode && (moveUp || moveDown)) velocity.y += (Number(moveUp) - Number(moveDown)) * speed * delta;

    const forward = new THREE.Vector3(); camera.getWorldDirection(forward); forward.y = 0; forward.normalize();
    const right = new THREE.Vector3().crossVectors(forward, camera.up).normalize();
    
    playerGroup.position.addScaledVector(forward, velocity.z * delta).addScaledVector(right, velocity.x * delta);
    playerGroup.position.y += velocity.y * delta;

    if (!isFlyMode) {
        raycaster.ray.origin.copy(playerGroup.position);
        raycaster.ray.origin.y += 2; 
        
        // 현재 눈에 보이는 물체만 충돌 테스트 대상으로 필터링
        const visibleObjects = objectsToTest.filter(obj => obj.visible);
        const intersections = raycaster.intersectObjects(visibleObjects);
        
        if (intersections.length > 0) {
            const groundHeight = intersections[0].point.y + PLAYER_HEIGHT;
            if (playerGroup.position.y < groundHeight) {
                velocity.y = 0;
                playerGroup.position.y = groundHeight;
                canJump = true;
            }
        }
    } else {
        if (playerGroup.position.y < 0.5) { playerGroup.position.y = 0.5; velocity.y = 0; }
    }

    if (!isThirdPerson) {
        camera.position.copy(playerGroup.position);
        const lookDir = new THREE.Vector3(); camera.getWorldDirection(lookDir);
        orbitControls.target.copy(camera.position).add(lookDir);
    } else {
        orbitControls.target.copy(playerGroup.position);
    }

    orbitControls.update();
    prevTime = time;
    renderer.render(scene, camera);
}

function onWindowResize() { 
    camera.aspect = window.innerWidth / window.innerHeight; 
    camera.updateProjectionMatrix(); 
    renderer.setSize(window.innerWidth, window.innerHeight); 
}
